Usage

Add ThemeProvider in root component

import { ThemeProvider } from '@flame-ui/react'
const { theme } = createTheme({
  colors: {
    primary: '#2672FF',
  },
})
const App = ({ children }) => {
  return <ThemeProvider theme={theme}>{children}</ThemeProvider>
}
